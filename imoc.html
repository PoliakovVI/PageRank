def plot_subplots(plot_x, plot_y, sp_x=1, sp_y=1, sp_id=1, title="", label="", linestyleid=0):
    linestyles = ["-", ":", "--", "-."]
    plt.subplot(sp_x, sp_y, sp_id)
    plt.title(title)
    plt.legend()
    plt.plot(plot_x, plot_y, linestyles[linestyleid], label=label)

kct_results = []
sq_results = []
dt_results = []
ids = []

plt.subplots_adjust(wspace=1, hspace=0.8)

for i in range(100):
    Generators.BAmodel(10, file_out="testfile.txt")
    bam_res = list(map(float, open("true_testfile.txt").readline().split()))
    tm = TransitionMatrix.TransitionMatrix()
    tm.read_from_txt("testfile.txt")
    pm = ComputingMethods.PowerMethod(tm)
    pm.stopping_run(precision=0.0001)
    baseline = pm._stat_vector
    
    ids.append(i)
    kct_results.append(Test.KendallCorrelationTest(bam_res, baseline))
    sq_results.append(Test.sequenceTest(bam_res, baseline))
    dt_results.append(Test.distanceTest(bam_res, baseline))

plot_subplots(ids, kct_results, title="KendallCorrelationTest", sp_x=3, sp_y=1, sp_id=1)
plot_subplots(ids, sq_results, title="sequenceTest", sp_x=3, sp_y=1, sp_id=2)
plot_subplots(ids, dt_results, title="distanceTest", sp_x=3, sp_y=1, sp_id=3)
