def plot_subplots(plot_x, plot_y, sp_x=1, sp_y=1, sp_id=1, title="", label=""):
    plt.subplot(sp_x, sp_y, sp_id)
    plt.title(title)
    plt.legend()
    plt.plot(plot_x, plot_y, label=label)
